## 回溯 + 剪枝

```js
const combine_1 = (n, k) => {
  const res = [], ans = [];
  const backTracking = (n, k, i) => {
    // 终止条件 ans.length == k
    if (ans.length == k) {
      res.push([...ans])
      return;
    }
    // 剪枝
    const p = n - ans.length + 1
    for (let j = i; j <= p; j++) {
      ans.push(j); // 处理节点
      backTracking(n, k, j + 1); // 递归遍历
      ans.pop() // 回溯操作
    }
  }
  backTracking(n, k, 1)
  return res;
}
```
