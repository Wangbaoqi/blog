**示例 1：**

```md
输入：n = 3
输出：["((()))","(()())","(())()","()(())","()()()"]
```

**示例 2：**

```md
输入：n = 1
输出：["()"]
```

**提示：**

- `1 <= n <= 8`


## 回溯

```js
const generateParenthesis_a = n => {
  const ret = [], ans = [];
  const backTracking = (max, l, r) => {
    if (ans.length == 2 * n) {
      ret.push(ans.join(''))
      return
    }
    if (l < max) {
      ans.push('(');
      backTracking(max, l + 1, r);
      ans.pop();
    }
    if (r < l) {
      ans.push(')');
      backTracking(max, l, r + 1);
      ans.pop();
    }
  }
  backTracking(n, 0, 0)
  return ret
}
```
